37 size_t find_node_index(const std::string &node_name)
38 {
39 auto it = std::find_if(nodes.begin(),
40 nodes.end(),
41 [&node_name](const NodeType &node)
42 { return !node.name.compare(node_name); });
43 if (it == nodes.end())

52 bool insert(NodeType node, const std::string &parent_name)
53 {
54 // No parent given - insert as root
55 if (!parent_name.size())
56 {
57 nodes.insert(nodes.begin(), node);
58 return true;
59 }
60
61 // Locate parent
62 auto parent_index = find_node_index(parent_name);
63 if (parent_index == EENG_NULL_INDEX)
64 return false;
65 auto pit = nodes.begin() + parent_index;
66
67 // Update branch strides within the same branch
68 // Iterate backwards from parent to the root and increment strides
69 // ranging to the insertion
70 //
71 auto prit = pit;
72 while (prit >= nodes.begin())
73 {
74 if (prit->m_branch_stride > (unsigned)std::distance(prit, pit))
75 prit->m_branch_stride++;
76 if (!prit->m_parent_ofs) // Discontinue past root (preceeding trees not affected)
77 break;
78 prit--;
79 }
80
81 // Update parent indices
82 // Iterate forward up until the next root,
83 // and increment parent offsets ranging to the insertion
84 //
85 auto pfit = pit + 1;
86 while (pfit < nodes.end())
87 {
88 if (!pfit->m_parent_ofs) // Discontinue at succeeding root
89 break;
90 if (pfit->m_parent_ofs >= (unsigned)std::distance(pit, pfit))
91 pfit->m_parent_ofs++;
92 pfit++;
93 }
94
95 // Insert new node after its parent
96 node.m_parent_ofs = 1;
97 pit->m_nbr_children++;
98 nodes.insert(pit + 1, node);
99
100 return true;
101 }
